125 unsigned Model::TextureFromFile(const char *path, const std::string &directory) {
126  std::string filename = std::string(path);
127  filename = directory + '/' + filename;
128 
129  unsigned int textureID;
130  glGenTextures(1, &textureID);
131 
132  int width, height, nrComponents;
133  unsigned char *data = stbi_load(filename.c_str(), &width, &height, &nrComponents, 0);
134  if (data)
135  {
136  GLenum format;
137  if (nrComponents == 1)
138  format = GL_RED;
139  else if (nrComponents == 3)
140  format = GL_RGB;
141  else if (nrComponents == 4)
142  format = GL_RGBA;
143 
144  glEnable(GL_BLEND);
145  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
146 
147  glBindTexture(GL_TEXTURE_2D, textureID);
148  glTexImage2D(GL_TEXTURE_2D, 0, GL_SRGB, width, height, 0, format, GL_UNSIGNED_BYTE, data);
149  glGenerateMipmap(GL_TEXTURE_2D);
150 
151  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
152  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
153  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
154  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
155 
156  stbi_image_free(data);
157  }
158  else
159  {
160  std::cout << "Texture failed to load at path: " << path << std::endl;
161  stbi_image_free(data);
162  }
163 
164  return textureID;
165 }
